import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller, acf, pacf
from statsmodels.tsa.arima.model import ARIMA
from google.colab import files
import warnings
warnings.filterwarnings("ignore")

# Step 1: Upload the Excel file in Colab
uploaded = files.upload()

# Step 2: Read the Excel file and extract PDI column
df = pd.read_excel('Tabla examen preg 18.xlsx')
pdi = df['PDI']

# Step 3: Apply natural logarithm to PDI and plot
log_pdi = np.log(pdi)
plt.figure(figsize=(10, 6))
plt.plot(log_pdi, label='Log(PDI)', color='blue')
plt.title('Log of Personal Disposable Income (1970.1 - 1991.4)', fontsize=14)
plt.xlabel('Quarter', fontsize=12)
plt.ylabel('Log(PDI)', fontsize=12)
plt.grid(True)
plt.legend()
plt.show()

# Step 4: Define ADF test function
def adf_test(series, title=''):
    print(f'\nADF Test: {title}')
    result = adfuller(series, autolag='AIC')
    print(f'ADF Statistic: {result[0]:.4f}')
    print(f'p-value: {result[1]:.4f}')
    print('Critical Values:')
    for key, value in result[4].items():
        print(f'   {key}: {value:.4f}')
    if result[1] < 0.05:
        print('Result: Stationary (reject null hypothesis)')
    else:
        print('Result: Non-stationary (fail to reject null hypothesis)')
    return result[1]

# Step 5: Run ADF test on log(PDI)
p_value = adf_test(log_pdi, 'Log(PDI)')

# Step 6: If non-stationary, compute first difference, plot, and re-run ADF test
if p_value >= 0.05:
    diff_log_pdi = log_pdi.diff().dropna()
    plt.figure(figsize=(10, 6))
    plt.plot(diff_log_pdi, label='First Difference of Log(PDI)', color='green')
    plt.title('First Difference of Log(PDI)', fontsize=14)
    plt.xlabel('Quarter', fontsize=12)
    plt.ylabel('Differenced Log(PDI)', fontsize=12)
    plt.grid(True)
    plt.legend()
    plt.show()
    adf_test(diff_log_pdi, 'First Difference of Log(PDI)')
    series_to_use = diff_log_pdi
    d = 1
else:
    series_to_use = log_pdi
    d = 0

# Step 7: Plot ACF and PACF of the series to be used
plt.figure(figsize=(12, 5))
plt.subplot(1, 2, 1)
acf_vals = acf(series_to_use, nlags=20)
plt.stem(range(21), acf_vals, linefmt='b-', markerfmt='bo', basefmt='r-')
plt.title('ACF', fontsize=14)
plt.xlabel('Lag', fontsize=12)
plt.ylabel('Autocorrelation', fontsize=12)
plt.grid(True)

plt.subplot(1, 2, 2)
pacf_vals = pacf(series_to_use, nlags=20)
plt.stem(range(21), pacf_vals, linefmt='b-', markerfmt='bo', basefmt='r-')
plt.title('PACF', fontsize=14)
plt.xlabel('Lag', fontsize=12)
plt.ylabel('Partial Autocorrelation', fontsize=12)
plt.grid(True)
plt.tight_layout()
plt.show()

# Step 8: Fit ARIMA model with (p=1, d, q=1)
model = ARIMA(log_pdi, order=(1, d, 1))
model_fit = model.fit()
print(model_fit.summary())

# Step 9: Plot actual vs fitted values
fitted = model_fit.fittedvalues
plt.figure(figsize=(10, 6))
plt.plot(log_pdi, label='Actual Log(PDI)', color='blue')
plt.plot(fitted, label='Fitted Values', color='red', linestyle='--')
plt.title('Actual vs Fitted Log(PDI) from ARIMA(1, {}, 1)'.format(d), fontsize=14)
plt.xlabel('Quarter', fontsize=12)
plt.ylabel('Log(PDI)', fontsize=12)
plt.grid(True)
plt.legend()
plt.show()
